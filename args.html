调用传参

调用端口

juno 1.0.1版本开始可以通过-p 指定端口 默认端口为50001

具体调用示例可以参考develop/example/client/client_test.go

调用传参


// 基于protobuf 3.0 语法规范

//传参信息
message YearningAuditOrder {  
    string SQL = 1;  //SQL语句
    string DataBase = 2;  //数据库
    string Table = 3;  //数据表
    bool Execute = 4;  //是否执行
    bool Check = 5;  //是否检测SQL
    bool IsDML = 6;  //是否DML语句
    bool Backup = 7; //是否备份(回滚)
    Source source = 8;  //目标数据源信息
    string WorkId = 9;  //工单id

}

message Source {    //目标数据源信息
    string Addr = 1;   //数据源地址
    string User = 2;   //数据源用户
    string Password = 3; //数据源密码
    int32 Port = 4; //数据源端口
}

//返回信息

//检测返回信息
message Record {
    string SQL = 1;
    int32 AffectRows = 2;
    string Status = 3;
    string Error = 4;
    int32 Level = 6;
}

message RecordSet {
    repeated Record record = 1;
}

//执行返回信息
message ExecOrder {
    string Message = 1;
}
